import 'package:flutter/material.dart'; import 'package:mobile_scanner/mobile_scanner.dart'; class QRScannerScreen extends StatefulWidget { const QRScannerScreen({super.key}); @override State createState() => _QRScannerScreenState(); } class _QRScannerScreenState extends State { final MobileScannerController controller = MobileScannerController( formats: const [ BarcodeFormat.code128, BarcodeFormat.code39, BarcodeFormat.ean13, BarcodeFormat.upcA, BarcodeFormat.itf, BarcodeFormat.qrCode, ], detectionSpeed: DetectionSpeed.normal, ); // Lista de IMEIs válidos (15 dígitos) detectados, en el orden en que aparecieron. // Guardamos hasta 2 — la mayoría de teléfonos dual SIM traen exactamente 2 pegados en la etiqueta. final List foundImeis = []; String? lastRejected; String? lastRejectedHint; bool confirmed = false; /// Intenta sacar un IMEI válido (15 dígitos) de lo que leyó el escáner. /// - Si es un número limpio de 15 dígitos → lo acepta tal cual (caso normal). /// - Si es un número larguísimo de puros dígitos (20+, como el EID de eSIM de Samsung, /// que tiene 32) → NO se toca; ahí no hay ningún IMEI real escondido adentro, sacar /// 15 dígitos cualquiera sería inventar un IMEI falso. Se avisa qué es en vez de aceptarlo. /// - Si mezcla el IMEI con texto/etiquetas (ej. "IMEI1:123456789012345") → ahí sí se /// extrae el tramo de 15 dígitos de adentro, porque ahí el IMEI real está presente. String? extractImei(String val) { final clean = val.replaceAll(RegExp(r'\s'), ''); if (RegExp(r'^\d{15}$').hasMatch(clean)) return clean; if (RegExp(r'^\d+$').hasMatch(clean) && clean.length >= 20) return null; // probable EID u otro ID largo final match = RegExp(r'\d{15}').firstMatch(clean); return match?.group(0); } void _onDetect(BarcodeCapture capture) { if (confirmed) return; bool changed = false; for (final barcode in capture.barcodes) { final val = barcode.rawValue ?? ''; if (val.isEmpty) continue; final clean = val.replaceAll(RegExp(r'\s'), ''); final imei = extractImei(clean); if (imei != null) { if (!foundImeis.contains(imei) && foundImeis.length < 2) { foundImeis.add(imei); changed = true; } } else if (foundImeis.isEmpty) { // Solo mostramos el "detectado pero no es IMEI" cuando aún no encontramos ningún IMEI válido, // para no distraer una vez que ya vamos por buen camino. lastRejected = clean.length > 20 ? clean.substring(0, 20) + '...' : clean; lastRejectedHint = (RegExp(r'^\d+$').hasMatch(clean) && clean.length >= 20) ? "Eso es el EID (identificador eSIM), no el IMEI — buscá la barra de código que dice 'IMEI 1'." : null; changed = true; } } if (changed) setState(() {}); } void _confirm() { if (foundImeis.isEmpty) return; confirmed = true; controller.stop(); // Si encontró 2, devolvemos una lista con ambos. Si encontró solo 1, devolvemos el String como antes // (mantiene compatibilidad con quien solo espera un IMEI). if (foundImeis.length >= 2) { Navigator.pop(context, List.from(foundImeis)); } else { Navigator.pop(context, foundImeis.first); } } void _reset() { setState(() { foundImeis.clear(); lastRejected = null; lastRejectedHint = null; }); } @override void dispose() { controller.dispose(); super.dispose(); } @override Widget build(BuildContext context) { return Scaffold( backgroundColor: Colors.black, appBar: AppBar( backgroundColor: Colors.black, title: const Text("Escanear IMEI", style: TextStyle(color: Colors.white, fontWeight: FontWeight.bold)), iconTheme: const IconThemeData(color: Colors.white), actions: [ IconButton( icon: const Icon(Icons.flash_on, color: Colors.white), onPressed: () => controller.toggleTorch(), tooltip: "Flash", ), ], ), body: Column( children: [ // Instrucción Container( width: double.infinity, color: const Color(0xFF1E293B), padding: const EdgeInsets.symmetric(horizontal: 16, vertical: 12), child: const Text( "📱 Apuntá al código de barras del IMEI\nEl IMEI tiene exactamente 15 dígitos\nSi hay 2 (dual SIM), los detecta ambos", textAlign: TextAlign.center, style: TextStyle(color: Colors.white70, fontSize: 13), ), ), // Cámara Expanded( flex: 3, child: Stack( children: [ MobileScanner(controller: controller, onDetect: _onDetect), // Marco guía — vertical y más alto, para abarcar los dos códigos de barra si están uno encima del otro Center( child: Container( width: double.infinity, height: 160, margin: const EdgeInsets.symmetric(horizontal: 30), decoration: BoxDecoration( border: Border.all(color: foundImeis.isNotEmpty ? const Color(0xFF4ADE80) : const Color(0xFF38BDF8), width: 3), borderRadius: BorderRadius.circular(12), ), ), ), ], ), ), // Panel inferior Expanded( flex: 2, child: Container( width: double.infinity, color: const Color(0xFF0F172A), padding: const EdgeInsets.all(20), child: SingleChildScrollView( child: Column( children: [ if (foundImeis.isNotEmpty) ...[ Text( foundImeis.length >= 2 ? "✅ 2 IMEIs detectados:" : "✅ IMEI detectado:", style: const TextStyle(color: Color(0xFF4ADE80), fontSize: 13, fontWeight: FontWeight.w600), ), const SizedBox(height: 8), ...foundImeis.asMap().entries.map((e) => Container( margin: const EdgeInsets.only(bottom: 8), padding: const EdgeInsets.all(12), decoration: BoxDecoration( color: const Color(0xFF1E293B), borderRadius: BorderRadius.circular(10), border: Border.all(color: const Color(0xFF4ADE80), width: 2), ), child: Column(children: [ Text( foundImeis.length >= 2 ? "IMEI ${e.key + 1}" : "IMEI", style: const TextStyle(color: Colors.grey, fontSize: 11), ), Text( e.value, style: const TextStyle( color: Color(0xFF4ADE80), fontSize: 20, fontWeight: FontWeight.w900, fontFamily: 'monospace', letterSpacing: 2, ), textAlign: TextAlign.center, ), ]), )), if (foundImeis.length == 1) const Padding( padding: EdgeInsets.only(bottom: 8), child: Text("Seguí apuntando si hay un segundo IMEI, o confirmá con este solo.", textAlign: TextAlign.center, style: TextStyle(color: Colors.grey, fontSize: 12)), ), Row(children: [ Expanded( child: OutlinedButton( onPressed: _reset, style: OutlinedButton.styleFrom(side: const BorderSide(color: Colors.grey)), child: const Text("Reintentar", style: TextStyle(color: Colors.grey)), ), ), const SizedBox(width: 12), Expanded( child: ElevatedButton( onPressed: _confirm, style: ElevatedButton.styleFrom( backgroundColor: const Color(0xFF4ADE80), foregroundColor: Colors.black, padding: const EdgeInsets.symmetric(vertical: 14), ), child: Text( foundImeis.length >= 2 ? "✓ Usar estos 2 IMEIs" : "✓ Usar este IMEI", style: const TextStyle(fontWeight: FontWeight.bold, fontSize: 15), ), ), ), ]), ] else ...[ if (lastRejected != null) ...[ Text("⚠️ Detectado pero no es IMEI:", style: TextStyle(color: Colors.amber.shade300, fontSize: 12)), const SizedBox(height: 4), Text(lastRejected!, style: const TextStyle(color: Colors.grey, fontSize: 12, fontFamily: 'monospace')), const SizedBox(height: 8), if (lastRejectedHint != null) Text(lastRejectedHint!, textAlign: TextAlign.center, style: const TextStyle(color: Color(0xFFFBBF24), fontSize: 12, fontWeight: FontWeight.w600)) else const Text("Seguí buscando el código de 15 dígitos...", style: TextStyle(color: Colors.grey, fontSize: 12)), ] else const Text("Apuntá la cámara al código de barras\nmás largo de la etiqueta", textAlign: TextAlign.center, style: TextStyle(color: Colors.grey, fontSize: 14)), const SizedBox(height: 16), // Tip visual Container( padding: const EdgeInsets.all(10), decoration: BoxDecoration(color: const Color(0x1938BDF8), borderRadius: BorderRadius.circular(10)), child: const Text( "💡 Tip: En la caja suelen estar así:\n" "• SN: número de serie (más corto)\n" "• IMEI 1 / IMEI 2: 15 dígitos cada uno\n" "Buscá los que dicen 'IMEI' al lado", style: TextStyle(color: Color(0xFF38BDF8), fontSize: 12), ), ), const SizedBox(height: 10), Container( padding: const EdgeInsets.all(10), decoration: BoxDecoration(color: const Color(0x19A78BFA), borderRadius: BorderRadius.circular(10)), child: const Text( "📱 En cajas de SAMSUNG: hay varios códigos juntos (código de producto, S/N, EID). " "Apuntá directo a las dos barras chiquitas marcadas 'A' y 'B' al pie de la etiqueta — " "esas son las únicas con el IMEI limpio.", style: TextStyle(color: Color(0xFFA78BFA), fontSize: 12), ), ), ], ], ), ), ), ), ], ), ); } }
